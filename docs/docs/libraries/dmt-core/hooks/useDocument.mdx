## Basic usage

````jsx
import { useDocument } from '@data-modelling-tool/core'

const [
  document,
  loading,
  updateDocument,
  hasError
] = useDocument(dataSourceId, documentId)

if (loading) {
  return <div>Loading...</div>
}

if (hasError) {
  return <div>Error getting the document</div>
}

<DisplayDocument document={document} />

````
